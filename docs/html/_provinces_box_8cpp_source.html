65void FProvincesBox::AddProvincePositions(UProvince* Province)
66{
67 const FVector2d& ProvinceLeftTopCorner = BoxesMap->GetLeftTopCorner(Province);
68 const FVector2d& ProvinceRightBottomCorner = BoxesMap->GetRightBottomCorner(Province);
69 LeftTopCorner.X = FMath::Min(LeftTopCorner.X, ProvinceLeftTopCorner.X);
70 LeftTopCorner.Y = FMath::Min(LeftTopCorner.Y, ProvinceLeftTopCorner.Y);
71 RightBottomCorner.X = FMath::Max(RightBottomCorner.X, ProvinceRightBottomCorner.X);
72 RightBottomCorner.Y = FMath::Max(RightBottomCorner.Y, ProvinceRightBottomCorner.Y);
73}

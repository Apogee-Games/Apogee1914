60bool UBoxesMap::AddProvinceToNeighbourBoxes(UProvince* Province)
61{
62 const UProvinceManager* ProvinceManager = GetGameInstance()->GetSubsystem<UProvinceManager>();
63 const TSet<FColor>& Neighbours = GetGameInstance()->GetSubsystem<UProvincesMap>()->GetNeighbourColors()[Province->GetId()];
64
65 TArray<TSharedPtr<FProvincesBox>> SameCountryBoxes;
66
67 for (const auto& NeighbourColor: Neighbours)
68 {
69 const UProvince* Neighbour = ProvinceManager->GetProvince(NeighbourColor);
70 if (!Neighbour) continue;
71 if (Province->GetCountryController() != Neighbour->GetCountryController()) continue;
72 if (ProvinceBox[Neighbour]->GetCountry() != Province->GetCountryController()) continue;
73 if (SameCountryBoxes.Contains(ProvinceBox[Neighbour])) continue;
74 SameCountryBoxes.Add(ProvinceBox[Neighbour]);
75 }
76
77 if (SameCountryBoxes.Num() == 1)
78 {
79 return AddProvinceToBox(SameCountryBoxes[0], Province);
80 }
81
82 if (SameCountryBoxes.Num() > 1)
83 {
84 return MergeBoxesAndAddProvince(SameCountryBoxes, Province);
85 }
86
87 return false;
88}

98bool UBoxesMap::MergeBoxesAndAddProvince(TArray<TSharedPtr<FProvincesBox>>& SameCountryBoxes, UProvince* Province)
99{
100 SameCountryBoxes.Sort(FProvincesBox::CompareAscending);
101 const int32 BoxesNumber = SameCountryBoxes.Num();
102 for (int32 i = 0; i < BoxesNumber - 1; ++i)
103 {
104 for (const auto& BoxProvince: SameCountryBoxes[i]->GetProvinces())
105 {
106 SameCountryBoxes[BoxesNumber - 1]->AddProvince(BoxProvince);
107 ProvinceBox[BoxProvince] = SameCountryBoxes[BoxesNumber - 1];
108 }
109 Boxes.Remove(SameCountryBoxes[i]);
110 }
111 AddProvinceToBox(SameCountryBoxes[BoxesNumber - 1], Province);
112 NotifyBoxUpdate(SameCountryBoxes[BoxesNumber - 1]);
113 return true;
114}

128void UBoxesMap::CalculateProvinceCorners()
129{
130 const UProvincesMap* ProvincesMap = GetGameInstance()->GetSubsystem<UProvincesMap>();
131 for (const auto& Province: GetGameInstance()->GetSubsystem<UProvinceManager>()->GetAllProvinces())
132 {
133
134 FVector2d LeftTopCorner = FVector2d(INT_MAX, INT_MAX);
135 FVector2d RightBottomCorner = FVector2d(0, 0);
136
137 for (const auto& Index: ProvincesMap->GetProvincePositions(Province->GetId()))
138 {
139 const FVector2d Position = FTextureUtils::GetPositionVector(Index, SizeVector);
140 LeftTopCorner.X = FMath::Min(LeftTopCorner.X, Position.X);
141 LeftTopCorner.Y = FMath::Min(LeftTopCorner.Y, Position.Y);
142 RightBottomCorner.X = FMath::Max(RightBottomCorner.X, Position.X);
143 RightBottomCorner.Y = FMath::Max(RightBottomCorner.Y, Position.Y);
144 }
145
146 LeftTopCorners.Add(Province, LeftTopCorner);
147 RightBottomCorners.Add(Province, RightBottomCorner);
148
149 }
150}

152void UBoxesMap::CalculateBoxes()
153{
154 UProvinceManager* ProvinceManager = GetGameInstance()->GetSubsystem<UProvinceManager>();
155 const TMap<FColor, TSet<FColor>>& Neighbours = GetGameInstance()->GetSubsystem<UProvincesMap>()->GetNeighbourColors();
156 for (const auto& [From, Tos]: Neighbours)
157 {
158 UProvince* Province = ProvinceManager->GetProvince(From);
159 if (!Province || ProvinceBox.Contains(Province)) continue;
160 TSharedPtr<FProvincesBox> Box = MakeShared<FProvincesBox>(this, Province->GetCountryController());
161 AddProvincesToBox(Box, Province, Province->GetCountryController(), ProvinceManager);
162 Boxes.Add(Box);

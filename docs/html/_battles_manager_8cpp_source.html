49void UBattlesManager::TickBattle(UProvince* Province, UUnitsMover* UnitsMover, UUnitsFactory* UnitsFactory)
50{
51 if (Province->GetAttackers().IsEmpty() || Province->GetDefenders().IsEmpty())
52 {
53 FinishedBattleProvinces.Enqueue(Province);
54 return;
55 }
56
57 const float AttackerAttackScore = CalculateAttackScore(Province->GetAttackers());
58 const float DefenderAttackScore = CalculateAttackScore(Province->GetDefenders());
59
60 TQueue<UUnit*> UnitsToRemove;
61
62 ApplyDamage(Province->GetAttackers(), DefenderAttackScore, UnitsToRemove, UnitsMover);
63 ApplyDamage(Province->GetDefenders(), AttackerAttackScore, UnitsToRemove, UnitsMover);
64
65 RemoveUnits(UnitsToRemove, UnitsFactory);
66}

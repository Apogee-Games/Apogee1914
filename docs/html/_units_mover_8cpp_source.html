135void UUnitsMover::MoveUnits()
136{
137 for (const auto& [Unit, Position]: Positions)
138 {
139 if (SuspendedUnits.Contains(Unit)) continue;
140 Paths[Unit][Position].Value--;
141 if (Paths[Unit][Position].Value == 0)
142 {
143 NotifyUnitMovement(Unit, Unit->GetPosition(), Paths[Unit][Position].Key);
144 Unit->Move(Paths[Unit][Position].Key);
145 Positions[Unit]++;
146 if (Positions[Unit] >= Paths[Unit].Num()) UnitsArrived.Enqueue(Unit);
147 }
148 }
